function countDayFromDate(date: string) {
  const today = new Date();
  const date2 = new Date(date);
  const diffTime = Math.abs(today.getTime() - date2.getTime());
  const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  return diffDays;
}
